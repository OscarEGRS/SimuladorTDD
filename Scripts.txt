Cuantos de un tipo de relacion hay


CREATE VIEW `CantidadTipoDeRelacion_view` AS
select count(relacion.TipoRelacion_idTipoRelacion) as Cantidad ,tiporelacion.NombreRelacion as Nombre from relacion
inner join tiporelacion on relacion.TipoRelacion_idTipoRelacion = tiporelacion.idTipoRelacion
group by relacion.TipoRelacion_idTipoRelacion

Cuantos de un tipo de relacion hay y nombre de los empleados

CREATE VIEW `CantidadTipoDeRelacion_view2` AS
select count(relacion.TipoRelacion_idTipoRelacion) as Cantidad ,tiporelacion.NombreRelacion as Nombre,empleados.NombreEmpleado  from relacion
inner join tiporelacion on relacion.TipoRelacion_idTipoRelacion = tiporelacion.idTipoRelacion
inner join empleados on relacion.Empleados_idEmpleados = empleados.idEmpleados
group by relacion.TipoRelacion_idTipoRelacion

tipos de relaciones y empleados 

CREATE VIEW `CantidadTipoDeRelacion_view3` AS
select tiporelacion.NombreRelacion as Nombre,empleados.NombreEmpleado, empleados2.NombreEmpleado as empleados2  from relacion
inner join tiporelacion on relacion.TipoRelacion_idTipoRelacion = tiporelacion.idTipoRelacion
inner join empleados on relacion.Empleados_idEmpleados = empleados.idEmpleados
inner join empleados as empleados2 on relacion.Empleados_idEmpleados1 = empleados2.idEmpleados


Relaciones de juan 

CREATE VIEW `CantidadTipoDeRelacion_view4` AS
select empleados.NombreEmpleado, tiporelacion.NombreRelacion, empleados2.NombreEmpleado as Empleado2  from relacion

inner join empleados on relacion.Empleados_idEmpleados = empleados.idEmpleados
inner join tiporelacion on relacion.TipoRelacion_idTipoRelacion = tiporelacion.idTipoRelacion
inner join empleados as empleados2 on relacion.Empleados_idEmpleados1 = empleados2.idEmpleados

where empleados.idEmpleados = '1'


Relaciones Con orden

CREATE VIEW `CantidadTipoDeRelacion_view5` AS
select empleados.NombreEmpleado, tiporelacion.NombreRelacion as Relacion, empleados2.NombreEmpleado as empleados2  from relacion
inner join tiporelacion on relacion.TipoRelacion_idTipoRelacion = tiporelacion.idTipoRelacion
inner join empleados on relacion.Empleados_idEmpleados = empleados.idEmpleados
inner join empleados as empleados2 on relacion.Empleados_idEmpleados1 = empleados2.idEmpleados
order by empleados2.NombreEmpleado,empleados.NombreEmpleado desc
